63/// Builder for converting variant values into strongly typed Arrow arrays.
64///
65/// Useful for variant_get kernels that need to extract specific paths from variant values, possibly
66/// with casting of leaf values to specific types.
67pub(crate) enum VariantToArrowRowBuilder<'a> {
68    Primitive(PrimitiveVariantToArrowRowBuilder<'a>),
69    BinaryVariant(VariantToBinaryVariantArrowRowBuilder),
70
71    // Path extraction wrapper - contains a boxed enum for any of the above
72    WithPath(VariantPathRowBuilder<'a>),
73}

187/// Creates a primitive row builder, returning Err if the requested data type is not primitive.
188pub(crate) fn make_primitive_variant_to_arrow_row_builder<'a>(
189    data_type: &'a DataType,
190    cast_options: &'a CastOptions,
191    capacity: usize,
192) -> Result<PrimitiveVariantToArrowRowBuilder<'a>> {
193    use PrimitiveVariantToArrowRowBuilder::*;
194
195    let builder =
196        match data_type {
197            DataType::Boolean => {
198                Boolean(VariantToBooleanArrowRowBuilder::new(cast_options, capacity))
199            }
200            DataType::Int8 => Int8(VariantToPrimitiveArrowRowBuilder::new(
201                cast_options,
202                capacity,
203            )),
204            DataType::Int16 => Int16(VariantToPrimitiveArrowRowBuilder::new(
205                cast_options,
206                capacity,
207            )),
208            DataType::Int32 => Int32(VariantToPrimitiveArrowRowBuilder::new(
209                cast_options,
210                capacity,
211            )),
212            DataType::Int64 => Int64(VariantToPrimitiveArrowRowBuilder::new(
213                cast_options,
214                capacity,
215            )),
216            DataType::UInt8 => UInt8(VariantToPrimitiveArrowRowBuilder::new(
217                cast_options,
218                capacity,
219            )),
220            DataType::UInt16 => UInt16(VariantToPrimitiveArrowRowBuilder::new(
221                cast_options,
222                capacity,
223            )),
224            DataType::UInt32 => UInt32(VariantToPrimitiveArrowRowBuilder::new(
225                cast_options,
226                capacity,
227            )),
228            DataType::UInt64 => UInt64(VariantToPrimitiveArrowRowBuilder::new(
229                cast_options,
230                capacity,
231            )),
232            DataType::Float16 => Float16(VariantToPrimitiveArrowRowBuilder::new(
233                cast_options,
234                capacity,
235            )),
236            DataType::Float32 => Float32(VariantToPrimitiveArrowRowBuilder::new(
237                cast_options,
238                capacity,
239            )),
240            DataType::Float64 => Float64(VariantToPrimitiveArrowRowBuilder::new(
241                cast_options,
242                capacity,
243            )),
244            DataType::Decimal32(precision, scale) => Decimal32(
245                VariantToDecimalArrowRowBuilder::new(cast_options, capacity, *precision, *scale)?,
246            ),
247            DataType::Decimal64(precision, scale) => Decimal64(
248                VariantToDecimalArrowRowBuilder::new(cast_options, capacity, *precision, *scale)?,
249            ),
250            DataType::Decimal128(precision, scale) => Decimal128(
251                VariantToDecimalArrowRowBuilder::new(cast_options, capacity, *precision, *scale)?,
252            ),
253            DataType::Decimal256(precision, scale) => Decimal256(
254                VariantToDecimalArrowRowBuilder::new(cast_options, capacity, *precision, *scale)?,
255            ),
256            DataType::Timestamp(TimeUnit::Microsecond, None) => TimestampMicroNtz(
257                VariantToTimestampNtzArrowRowBuilder::new(cast_options, capacity),
258            ),
259            DataType::Timestamp(TimeUnit::Microsecond, tz) => TimestampMicro(
260                VariantToTimestampArrowRowBuilder::new(cast_options, capacity, tz.clone()),
261            ),
262            DataType::Timestamp(TimeUnit::Nanosecond, None) => TimestampNanoNtz(
263                VariantToTimestampNtzArrowRowBuilder::new(cast_options, capacity),
264            ),
265            DataType::Timestamp(TimeUnit::Nanosecond, tz) => TimestampNano(
266                VariantToTimestampArrowRowBuilder::new(cast_options, capacity, tz.clone()),
267            ),
268            DataType::Date32 => Date(VariantToPrimitiveArrowRowBuilder::new(
269                cast_options,
270                capacity,
271            )),
272            _ if data_type.is_primitive() => {
273                return Err(ArrowError::NotYetImplemented(format!(
274                    "Primitive data_type {data_type:?} not yet implemented"
275                )));
276            }
277            _ => {
278                return Err(ArrowError::InvalidArgumentError(format!(
279                    "Not a primitive type: {data_type:?}"
280                )));
281            }
282        };
283    Ok(builder)
284}

286pub(crate) fn make_variant_to_arrow_row_builder<'a>(
287    metadata: &BinaryViewArray,
288    path: VariantPath<'a>,
289    data_type: Option<&'a DataType>,
290    cast_options: &'a CastOptions,
291    capacity: usize,
292) -> Result<VariantToArrowRowBuilder<'a>> {
293    use VariantToArrowRowBuilder::*;
294
295    let mut builder = match data_type {
296        // If no data type was requested, build an unshredded VariantArray.
297        None => BinaryVariant(VariantToBinaryVariantArrowRowBuilder::new(
298            metadata.clone(),
299            capacity,
300        )),
301        Some(DataType::Struct(_)) => {
302            return Err(ArrowError::NotYetImplemented(
303                "Converting unshredded variant objects to arrow structs".to_string(),
304            ));
305        }
306        Some(
307            DataType::List(_)
308            | DataType::LargeList(_)
309            | DataType::ListView(_)
310            | DataType::LargeListView(_)
311            | DataType::FixedSizeList(..),
312        ) => {
313            return Err(ArrowError::NotYetImplemented(
314                "Converting unshredded variant arrays to arrow lists".to_string(),
315            ));
316        }
317        Some(data_type) => {
318            let builder =
319                make_primitive_variant_to_arrow_row_builder(data_type, cast_options, capacity)?;
320            Primitive(builder)
321        }
322    };
323
324    // Wrap with path extraction if needed
325    if !path.is_empty() {
326        builder = WithPath(VariantPathRowBuilder {
327            builder: Box::new(builder),
328            path,
329        })
330    };
331
332    Ok(builder)
333}

335/// A thin wrapper whose only job is to extract a specific path from a variant value and pass the
336/// result to a nested builder.
337pub(crate) struct VariantPathRowBuilder<'a> {
338    builder: Box<VariantToArrowRowBuilder<'a>>,
339    path: VariantPath<'a>,
340}
341
342impl<'a> VariantPathRowBuilder<'a> {
343    fn append_null(&mut self) -> Result<()> {
344        self.builder.append_null()
345    }
346
347    fn append_value(&mut self, value: Variant<'_, '_>) -> Result<bool> {
348        if let Some(v) = value.get_path(&self.path) {
349            self.builder.append_value(v)
350        } else {
351            self.builder.append_null()?;
352            Ok(false)
353        }
354    }
355
356    fn finish(self) -> Result<ArrayRef> {
357        self.builder.finish()
358    }
359}

361macro_rules! define_variant_to_primitive_builder {
362    (struct $name:ident<$lifetime:lifetime $(, $generic:ident: $bound:path )?>
363    |$array_param:ident $(, $field:ident: $field_type:ty)?| -> $builder_name:ident $(< $array_type:ty >)? { $init_expr: expr },
364    |$value: ident| $value_transform:expr,
365    type_name: $type_name:expr) => {
366        pub(crate) struct $name<$lifetime $(, $generic : $bound )?>
367        {
368            builder: $builder_name $(<$array_type>)?,
369            cast_options: &$lifetime CastOptions<$lifetime>,
370        }
371
372        impl<$lifetime $(, $generic: $bound+ )?> $name<$lifetime $(, $generic )?> {
373            fn new(
374                cast_options: &$lifetime CastOptions<$lifetime>,
375                $array_param: usize,
376                // add this so that $init_expr can use it
377                $( $field: $field_type, )?
378            ) -> Self {
379                Self {
380                    builder: $init_expr,
381                    cast_options,
382                }
383            }
384
385            fn append_null(&mut self) -> Result<()> {
386                self.builder.append_null();
387                Ok(())
388            }
389
390            fn append_value(&mut self, $value: &Variant<'_, '_>) -> Result<bool> {
391                if let Some(v) = $value_transform {
392                    self.builder.append_value(v);
393                    Ok(true)
394                } else {
395                    if !self.cast_options.safe {
396                        // Unsafe casting: return error on conversion failure
397                        return Err(ArrowError::CastError(format!(
398                            "Failed to extract primitive of type {} from variant {:?} at path VariantPath([])",
399                            $type_name,
400                            $value
401                        )));
402                    }
403                    // Safe casting: append null on conversion failure
404                    self.builder.append_null();
405                    Ok(false)
406                }
407            }
408
409            fn finish(mut self) -> Result<ArrayRef> {
410                Ok(Arc::new(self.builder.finish()))
411            }
412        }
413    }
414}

446/// Builder for converting variant values to arrow Decimal values
447pub(crate) struct VariantToDecimalArrowRowBuilder<'a, T>
448where
449    T: DecimalType,
450    T::Native: DecimalCast,
451{
452    builder: PrimitiveBuilder<T>,
453    cast_options: &'a CastOptions<'a>,
454    precision: u8,
455    scale: i8,
456}
457
458impl<'a, T> VariantToDecimalArrowRowBuilder<'a, T>
459where
460    T: DecimalType,
461    T::Native: DecimalCast,
462{
463    fn new(
464        cast_options: &'a CastOptions<'a>,
465        capacity: usize,
466        precision: u8,
467        scale: i8,
468    ) -> Result<Self> {
469        let builder = PrimitiveBuilder::<T>::with_capacity(capacity)
470            .with_precision_and_scale(precision, scale)?;
471        Ok(Self {
472            builder,
473            cast_options,
474            precision,
475            scale,
476        })
477    }
478
479    fn append_null(&mut self) -> Result<()> {
480        self.builder.append_null();
481        Ok(())
482    }
483
484    fn append_value(&mut self, value: &Variant<'_, '_>) -> Result<bool> {
485        if let Some(scaled) = variant_to_unscaled_decimal::<T>(value, self.precision, self.scale) {
486            self.builder.append_value(scaled);
487            Ok(true)
488        } else if self.cast_options.safe {
489            self.builder.append_null();
490            Ok(false)
491        } else {
492            Err(ArrowError::CastError(format!(
493                "Failed to cast to {}(precision={}, scale={}) from variant {:?}",
494                T::PREFIX,
495                self.precision,
496                self.scale,
497                value
498            )))
499        }
500    }
501
502    fn finish(mut self) -> Result<ArrayRef> {
503        Ok(Arc::new(self.builder.finish()))
504    }
505}

507/// Builder for creating VariantArray output (for path extraction without type conversion)
508pub(crate) struct VariantToBinaryVariantArrowRowBuilder {
509    metadata: BinaryViewArray,
510    builder: VariantValueArrayBuilder,
511    nulls: NullBufferBuilder,
512}
513
514impl VariantToBinaryVariantArrowRowBuilder {
515    fn new(metadata: BinaryViewArray, capacity: usize) -> Self {
516        Self {
517            metadata,
518            builder: VariantValueArrayBuilder::new(capacity),
519            nulls: NullBufferBuilder::new(capacity),
520        }
521    }
522}
523
524impl VariantToBinaryVariantArrowRowBuilder {
525    fn append_null(&mut self) -> Result<()> {
526        self.builder.append_null();
527        self.nulls.append_null();
528        Ok(())
529    }
530
531    fn append_value(&mut self, value: Variant<'_, '_>) -> Result<bool> {
532        self.builder.append_value(value);
533        self.nulls.append_non_null();
534        Ok(true)
535    }
536
537    fn finish(mut self) -> Result<ArrayRef> {
538        let variant_array = VariantArray::from_parts(
539            self.metadata,
540            Some(self.builder.build()?),
541            None, // no typed_value column
542            self.nulls.finish(),
543        );
544
545        Ok(ArrayRef::from(variant_array))
546    }
